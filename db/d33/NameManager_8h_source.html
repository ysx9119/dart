90  bool setPattern(const std::string& _newPattern)
91  {
92  size_t name_start = _newPattern.find("%s");
93  size_t number_start = _newPattern.find("%d");
94 
95  if(name_start == std::string::npos || number_start == std::string::npos)
96  return false;
97 
98  if(name_start < number_start)
99  mNameBeforeNumber = true;
100  else
101  mNameBeforeNumber = false;
102 
103  size_t prefix_end = std::min(name_start, number_start);
104  size_t infix_end = std::max(name_start, number_start);
105 
106  mPrefix = _newPattern.substr(0, prefix_end);
107  mInfix = _newPattern.substr(prefix_end+2, infix_end-prefix_end-2);
108  mAffix = _newPattern.substr(infix_end+2);
109 
110  return true;
111  }

114  std::string issueNewName(const std::string& _name) const
115  {
116  if(!hasName(_name))
117  return _name;
118 
119  int count = 1;
120  std::string newName;
121  do
122  {
123  std::stringstream ss;
124  if(mNameBeforeNumber)
125  ss << mPrefix << _name << mInfix << count++ << mAffix;
126  else
127  ss << mPrefix << count++ << mInfix << _name << mAffix;
128  newName = ss.str();
129  } while (hasName(newName));
130 
131  dtwarn << "The name '" << _name << "' is a duplicate, "
132  << "so it has been renamed to '" << newName << "'\n";
133 
134  return newName;
135  }
